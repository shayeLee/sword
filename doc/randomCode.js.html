    /**
 * @namespace randomCode
 */
const randomCode = {
  /**
   * 生成指定长度的随机码
   * @memberof randomCode
   * @function generate
   * @param {number} length - 随机码的长度，必须大于零
   * @returns {string}
  */
  generate: function (length) {
    if (length <= 0) return null;

    var len = length;
    var chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789";
    var maxPos = chars.length;
    var code = "";
    for(var i = 0; i < len; i++) {
      code += chars.charAt(Math.floor(Math.random() * maxPos));
    }
    return code;
  },

  /**
   * 生成OSS文件上传路径
   * @memberof randomCode
   * @function generateOSSKey
   * @param {string} randomCode - 12位随机码
   * @param {string} name - 带后缀的文件名
   * @returns {string}
  */
  generateOSSKey: function(randomCode, name) {
    const time = new Date();
    const rootPath = `live/tms/${time.getFullYear()}/${time.getMonth() + 1}/${time.getDate()}/`;
    return `${rootPath}${randomCode}-${name}`;
  }
};

export default randomCode;
